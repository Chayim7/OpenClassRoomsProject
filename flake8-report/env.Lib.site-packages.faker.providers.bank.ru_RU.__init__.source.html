  • E501 Line too long (95 > 79 characters)
731 department_code: str = self.numerify(self.random_element(self.department_code_formats))
  • E501 Line too long (113 > 79 characters)
732 credit_organization_code: str = self.numerify(self.random_element(self.credit_organization_code_formats))
733 return "04" + region + department_code + credit_organization_code
