6 def get_clabe_control_digit(clabe: str) -> int:
7 """Generate the checksum digit for a CLABE.
8  
9 :param clabe: CLABE.
10 :return: The CLABE checksum digit.
11 """
12 factors = [3, 7, 1]
13 products: List[int] = []
14  
15 for i, digit in enumerate(clabe[:17]):
16 products.append((int(digit) * factors[i % 3]) % 10)
17  
18 return (10 - sum(products)) % 10

22 """Check if a CLABE is valid using the checksum.
